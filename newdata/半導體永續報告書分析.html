TSMC_clean_text <- function(page_text) {
  # 定義要移除的關鍵詞
  patterns <- c("112 年度永續報告書", "導言", "實踐永續管理", "營運與治理", "附錄", 
                "追求創新的先行者", "負責任的採購者", "綠色力量的執行者", "員工引以為傲的公司", "改變社會的力量")
  
  # 遍歷每個關鍵詞並替換為空白
  for (pattern in patterns) {
    page_text <- gsub(pattern, "", page_text)
  }
  
  # 移除所有空白字元
  page_text <- gsub("\\s+", "", page_text)
  
  return(page_text)
}
